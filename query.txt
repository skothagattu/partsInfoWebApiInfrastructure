SELECT DISTINCTROW [CMI/DESC].[CMI NO], [CMI/DESC].DESCRIPTION, [CMI/VENDOR].CODE, [CMI/VENDOR].[MFG NO], [CMI/VENDOR].SUB, [CMI/VENDOR].Date, [CMI/VENDOR].NOTES
FROM [CMI/DESC] LEFT JOIN [CMI/VENDOR] ON [CMI/DESC].[CMI NO] = [CMI/VENDOR].[CMI NO]
WHERE ((([CMI/VENDOR].Date) Is Not Null And ([CMI/VENDOR].Date)>[ENTER DATE]))
ORDER BY [CMI/DESC].[CMI NO];
